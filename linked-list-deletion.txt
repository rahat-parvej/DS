#include <stdio.h>
#include <stdlib.h>

struct Node
{
    int data;
    struct Node* next;
};

void append(struct Node** head_ref, int new_data)
{
    struct Node* new_node = (struct Node*) malloc(sizeof(struct Node));
    struct Node *last = *head_ref;
    new_node->data = new_data;
    new_node->next = NULL;
    if (*head_ref == NULL)
    {
        *head_ref = new_node;
        return;
    }
    while (last->next != NULL)
        last = last->next;
    last->next = new_node;
    return;
}


void deleteNode(struct Node** head_ref, int key)
{
    struct Node *temp = *head_ref, *prev;

    if (temp != NULL && temp->data == key)
    {
        *head_ref = temp->next;
        free(temp);
        return;
    }

    while (temp != NULL && temp->data != key)
    {
        prev = temp;
        temp = temp->next;
    }

    if (temp == NULL)
        return;

    prev->next = temp->next;

    free(temp);
}

void printList(struct Node* node)
{
    while (node != NULL)
    {
        printf("%d ", node->data);
        node = node->next;
    }
}

int main()
{
    struct Node* head = NULL;

    int i,x,n,opt=0;

    while(opt!=3)
    {
        printf("Enter number for action: 1.Insert 2.Delete 3.Exit\n");
        scanf("%d",&opt);
        if(opt==1)
        {
            printf("Enter number of elements for input: \n");
            scanf("%d",&n);

            printf("Enter inputs: \n");
            for(int i=0; i<n; i++)
            {
                scanf("%d",&x);
                append(&head,x);
            }

            puts("Created Linked List:");
            printList(head);
            printf("\n");
        }
        else if(opt==2)
        {
            printf("Enter Node to delete\n");
            scanf("%d",&x);

            deleteNode(&head, x);
            printf("Linked List after Deletion : ");
            printList(head);
            printf("\n");
        }

    }
    return 0;
}
